  文档解析与测试合成系统 - 实现计划（修订版）

  一、背景与目标

  1.1 需求背景

  现有项目是一个Agent测试框架（用于测试tasks/中的新型agent代码，并与ReAct等基线方法比较）。

  新需求的定位：为测试框架提供四个独立的基础服务模块：
  1. API语义归一化服务 - 从Swagger生成标准化的API能力模型
  2. API拓扑图服务 - 构建和查询API依赖关系图
  3. 测试场景生成服务 - 基于API路径逆向生成业务测试问题
  4. 有状态Mock服务 - 提供带状态管理和混沌注入的Mock API环境

  这些模块独立部署，可被测试框架或其他系统调用。

  1.2 架构定位

  Flora-evaluation/
  ├── tasks/                          # 现有：Agent核心代码

  │
  ├── api_normalization/              # 新增：阶段一服务

  2.18 内部实现

  1. MockServer (server.py)
    from fastapi import FastAPI, HTTPException
    import uvicorn

  class MockServer:
      """动态Mock服务器"""

      def __init__(self, state_manager, chaos_engine):
          self.app = FastAPI()
          self.state_manager = state_manager
          self.chaos_engine = chaos_engine
    
      def register_routes(self, api_definitions: List[dict]):
          """根据API定义动态注册路由"""
    
          for api in api_definitions:
              self._create_route(api)
    
      def _create_route(self, api_def: dict):
          """创建单个路由"""
    
          @self.app.api_route(
              path=api_def["path"],
              methods=[api_def["method"]]
          )
          async def dynamic_handler(**kwargs):
              # 1. 混沌注入检查
              if self.chaos_engine.should_inject():
                  return self.chaos_engine.inject()
    
              # 2. 状态操作
              if api_def["method"] == "POST":
                  return self.state_manager.create(api_def["entity"], kwargs)
              elif api_def["method"] == "GET":
                  return self.state_manager.get(api_def["entity"], kwargs.get("id"))
              elif api_def["method"] == "PUT":
                  return self.state_manager.update(api_def["entity"], kwargs.get("id"), kwargs)
              elif api_def["method"] == "DELETE":
                  return self.state_manager.delete(api_def["entity"], kwargs.get("id"))

  2. StateManager (state_manager.py)
    import sqlite3

  class StateManager:
      """基于SQLite的状态管理"""

      def __init__(self, db_path: str = ":memory:"):
          self.conn = sqlite3.connect(db_path, check_same_thread=False)
          self.entities = {}  # {entity_type: {id: data}}
    
      def create(self, entity_type: str, data: dict) -> dict:
          """创建实体"""
          entity_id = str(uuid.uuid4())
          if entity_type not in self.entities:
              self.entities[entity_type] = {}
          self.entities[entity_type][entity_id] = data
          return {"id": entity_id, **data}
    
      def get(self, entity_type: str, entity_id: str) -> dict:
          """获取实体"""
          if entity_type not in self.entities:
              raise HTTPException(404, f"{entity_type} not found")
          if entity_id not in self.entities[entity_type]:
              raise HTTPException(404, f"{entity_type} {entity_id} not found")
          return self.entities[entity_type][entity_id]
    
      # update, delete 类似实现

  3. ChaosEngine (chaos_engine.py)
    class ChaosEngine:
      """混沌注入引擎"""

      def __init__(self, config: dict):
          self.format_shake_rate = config.get("format_shake_rate", 0.0)
          self.latency_rate = config.get("latency_rate", 0.0)
          self.error_rate = config.get("error_rate", 0.0)

      def should_inject(self) -> bool:
          """判断是否注入"""
          return random.random() < (self.format_shake_rate + self.latency_rate + self.error_rate)

      def inject(self) -> dict:
          """执行注入"""
          chaos_type = random.choices(
              ["format_shake", "latency", "error"],
              weights=[self.format_shake_rate, self.latency_rate, self.error_rate]
          )[0]

          if chaos_type == "format_shake":
              # 随机切换命名风格
              return self._shake_format()
          elif chaos_type == "latency":
              time.sleep(random.uniform(0.5, 3.0))
              return {"status": "ok"}
          else:
              raise HTTPException(500, "Internal Server Error")
